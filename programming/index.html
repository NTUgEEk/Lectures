Decorators

A function of function to decorate functions.

Generators

What does range(10) in python does?
It throws out 1, 2, 3 ... in order. That's what generator does. 

def range(a):
  t = 0
  while t < a:
    yield t
    t += 1

gen = range(3)
next(gen) # 0
next(gen) # 1
            

function* range(a) {
  let t = 0;
  while (t < a) {
    yield t;
    t += 1;
  }
}

let gen = range(3);
gen.next().value // 0
gen.next().value // 1

Generators

Can also send in value


def range(a):
  t = 0
  while t < a:
    t += yield t

gen = range(40)
gen.send(None) # 0
gen.send(10) # 10
gen.send(20) # 30
            

function* range(a) {
  let t = 0;
  while (t < a) {
    t += yield t;
  }
}

let gen = range(3);
gen.next(10).value // 10
gen.next(20).value // 30
